import configurator, os
p = configurator.start_it_up(getBundlePath(), "prefab_locator_semaphore.blend")
try:
    mouseMove(Location(0,0)); wait(, 5)
    click(Pattern().similar(0.95).targetOffset(53,0))
    click(Pattern().similar(0.80))
    click(Pattern().similar(0.95).targetOffset(53,0)); type("crossi")
    click(Pattern().similar(0.90))
    click(Pattern().similar(0.90).targetOffset(54,1))
    click(Pattern().similar(0.90))
    find(Pattern().similar(0.90))
    click(Pattern().similar(0.90).targetOffset(55,1))
    click(Pattern().similar(0.90))
    click(Pattern().targetOffset(-61,-11)); type("12" + Key.ENTER)  # Change Green light value
    click(Pattern().targetOffset(2,-11)); type("2.5" + Key.ENTER)  # Change Orange light value
    click(Pattern().targetOffset(62,-11)); type("22" + Key.ENTER)  # Change Red light value
    click(Pattern().targetOffset(126,-11)); type("1.5" + Key.ENTER)  # Change after-red Orange light value
    click(Pattern().targetOffset(0,11)); type("3" + Key.ENTER)  # Change Cycle Delay value
    hover(Location(400, 400))  # move cursor away from panels
    find(Pattern().similar(0.95))
    click(Pattern().similar(0.95).targetOffset(48,0))
    click(Pattern().similar(0.90))
    click(Pattern().targetOffset(-37,-11)); type("160" + Key.ENTER)  # Change Green light value
    click(Pattern().targetOffset(37,-11)); type("200" + Key.ENTER)  # Change Orange light value
    click(Pattern().targetOffset(117,-11)); type("100" + Key.ENTER)  # Change Red light value
    click(Pattern().targetOffset(0,11)); type("2" + Key.ENTER)  # Change Cycle Delay value
    hover(Location(400, 400))  # move cursor away from panels
    find(Pattern().similar(0.95))
except:
    configurator.save_screenshot(getBundlePath(), Screen())
    raise
finally:
    configurator.close_blender(p)
