import configurator, os
p = configurator.start_it_up(getBundlePath(), "prefab_locator_trigger.blend")
try:
    click(Pattern().similar(0.90))  # get rid of 3D cursor
    wait(, 5)
    hover(Location(400, 400))  # move cursor above 3D view

    type("g0" + Key.TAB + "0" + Key.TAB + ".1" + Key.ENTER)  # move locator

    keyDown(Key.SHIFT); type("d"); keyUp(Key.SHIFT)  # copy locator
    type("1" + Key.TAB + "2" + Key.TAB + "0" + Key.ENTER)  # move locator

    keyDown(Key.SHIFT); type("d"); keyUp(Key.SHIFT)  # copy locator
    type("2-" + Key.TAB + "0" + Key.TAB + "0" + Key.ENTER)  # move locator

    rightClick(Pattern().similar(0.90))
    keyDown(Key.SHIFT); rightClick(Pattern().similar(0.90).targetOffset(54,-39)); keyUp(Key.SHIFT)
    click()
    rightClick(Pattern().similar(0.90).targetOffset(63,-40))
    keyDown(Key.SHIFT); rightClick(Pattern().similar(0.90).targetOffset(8,-41)); keyUp(Key.SHIFT)
    click()
    find(Pattern().similar(0.90))  # check the result
    rightClick(Pattern().similar(0.90).targetOffset(8,-41))
    keyDown(Key.SHIFT); rightClick(Pattern().similar(0.90).targetOffset(-56,1)); keyUp(Key.SHIFT)
    click()
    rightClick(Pattern().similar(0.90).targetOffset(-86,21))  # check the result in the same time
    click(Pattern().similar(0.90)); type("1.5" + Key.ENTER)
    rightClick(Pattern().similar(0.90).targetOffset(-17,6))
    click(Pattern().similar(0.90)); type("1" + Key.ENTER)
    rightClick(Pattern().similar(0.90).targetOffset(0,28))
    click(Pattern().similar(0.90)); type("1.3" + Key.ENTER)
    find()
except:
    configurator.save_screenshot(getBundlePath(), Screen())
    raise
finally:
    configurator.close_blender(p)
